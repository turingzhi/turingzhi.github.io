    my_list = [10, 20, 30]

# Add operations
my_list.append(40)       # Adds 40 at the end
my_list.insert(1, 15)    # Inserts 15 at index 1
my_list.extend([50, 60]) # Adds 50 and 60 at the end

# Remove operations
my_list.remove(20)  # Removes the first occurrence of 20
my_list.pop(2)      # Removes the element at index 2
del my_list[1]      # Deletes the element at index 1

print(my_list)  # Output after operations

    It is crucial to have : inside the [] when using remove or pop method. Otherwise, it will raise an error which is skipping some arguments.

        [(600, 600), (600, 600), (600, 600)]
        >>> len(player2.deck.cards)
        0
        """
        orig_opponent_deck_length = len(opponent.deck.cards)
